---
title: "fee.created"
description: "Published when a new usage-based fee is created."
---

## Event Details

| Property | Value |
|----------|-------|
| **Type** | `fee.created` |

## Payload Schema

<ResponseField name="feeId" type="string (UUID v7)" required>
  UUID v7 identifier.
</ResponseField>

<ResponseField name="externalProductRef" type="string" required>
  External product reference.
</ResponseField>

<ResponseField name="externalPriceRef" type="string" required>
  External price reference.
</ResponseField>

<ResponseField name="externalBillingMeterRef" type="string" required>
  External billing meter reference.
</ResponseField>

<ResponseField name="name" type="string" required>
  Fee name.
</ResponseField>

<ResponseField name="description" type="string">
  Fee description.
</ResponseField>

<ResponseField name="eventName" type="string" required>
  Unique event name associated with the fee.
</ResponseField>

<ResponseField name="currency" type="string" required>
  Fee currency (`USD`, `BRL`, `EUR`).
</ResponseField>

<ResponseField name="status" type="string" required>
  Fee status: `PENDING`, `ACTIVE`, or `INACTIVE`.
</ResponseField>

<ResponseField name="createdBy" type="string (UUID)" required>
  User who created the fee.
</ResponseField>

<ResponseField name="createdAt" type="string (ISO 8601)" required>
  Creation timestamp.
</ResponseField>

<ResponseField name="updatedBy" type="string (UUID)" required>
  User who last updated the fee.
</ResponseField>

<ResponseField name="updatedAt" type="string (ISO 8601)" required>
  Last update timestamp.
</ResponseField>
