---
title: "invoice.paid"
description: "Published when an invoice is successfully paid."
---

## Event Details

| Property | Value |
|----------|-------|
| **Type** | `invoice.paid` |

## Payload Schema

<ResponseField name="invoiceId" type="string (UUID v7)" required>
  Invoice identifier.
</ResponseField>

<ResponseField name="organizationId" type="string (UUID v7)" required>
  FK to Organization.
</ResponseField>

<ResponseField name="subscriptionId" type="string (UUID v7)">
  FK to Subscription.
</ResponseField>

<ResponseField name="externalRef" type="string" required>
  External invoice reference.
</ResponseField>

<ResponseField name="type" type="string" required>
  Invoice type: `PLAN` or `FEE`.
</ResponseField>

<ResponseField name="paymentMethod" type="string" required>
  Payment method: `CARD` or `BOLETO`.
</ResponseField>

<ResponseField name="card" type="object">
  Card details: `{brand, last4, expMonth, expYear}`.
</ResponseField>

<ResponseField name="boleto" type="object">
  Boleto details: `{url}`.
</ResponseField>

<ResponseField name="currency" type="string" required>
  Currency code (`USD`, `BRL`, `EUR`).
</ResponseField>

<ResponseField name="totalAmount" type="integer" required>
  Total in cents.
</ResponseField>

<ResponseField name="discountAmount" type="integer" required>
  Discount in cents.
</ResponseField>

<ResponseField name="payableAmount" type="integer" required>
  Payable in cents.
</ResponseField>

<ResponseField name="discounts" type="array" required>
  Applied discounts.
</ResponseField>

<ResponseField name="lines" type="array" required>
  Invoice line items.
</ResponseField>

<ResponseField name="status" type="string" required>
  Invoice status.
</ResponseField>

<ResponseField name="issuedAt" type="string (ISO 8601)" required>
  Issue date.
</ResponseField>

<ResponseField name="dueDate" type="string (ISO 8601)" required>
  Due date.
</ResponseField>

<ResponseField name="paidAt" type="string (ISO 8601)" required>
  Payment timestamp.
</ResponseField>

<ResponseField name="createdAt" type="string (ISO 8601)" required>
  Creation timestamp.
</ResponseField>

<ResponseField name="updatedAt" type="string (ISO 8601)" required>
  Last update timestamp.
</ResponseField>
