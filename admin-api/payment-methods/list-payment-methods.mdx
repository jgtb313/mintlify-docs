---
title: "List Payment Methods"
api: "GET /organizations/{organizationId}/payment-methods"
description: "List payment methods."
---
<Info>Requires `payment_method:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>Organization UUID.</ParamField>

## Response

<ResponseField name="data" type="array">
  List of payment methods.
  <Expandable title="payment method properties">
    <ResponseField name="paymentMethodId" type="string">Payment method ID.</ResponseField>
    <ResponseField name="organizationId" type="string">Organization UUID.</ResponseField>
    <ResponseField name="type" type="string">Payment method type (`CARD` or `BOLETO`).</ResponseField>
    <ResponseField name="brand" type="string">Card brand (CARD only).</ResponseField>
    <ResponseField name="last4" type="string">Last 4 digits (CARD only).</ResponseField>
    <ResponseField name="expMonth" type="integer">Expiration month (CARD only).</ResponseField>
    <ResponseField name="expYear" type="integer">Expiration year (CARD only).</ResponseField>
    <ResponseField name="isDefault" type="boolean">Whether it is the default payment method.</ResponseField>
    <ResponseField name="billingDetails" type="object">
      Billing details.
      <Expandable title="billingDetails properties">
        <ResponseField name="name" type="string">Billing name.</ResponseField>
        <ResponseField name="email" type="string">Billing email.</ResponseField>
        <ResponseField name="phone" type="string">Billing phone.</ResponseField>
        <ResponseField name="address" type="object">Billing address.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
