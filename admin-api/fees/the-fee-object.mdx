---
title: "The Fee Object"
---

<ResponseField name="feeId" type="string">
  The fee unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="externalProductRef" type="string">
  The external product reference in the payment provider.
</ResponseField>

<ResponseField name="externalPriceRef" type="string">
  The external price reference in the payment provider.
</ResponseField>

<ResponseField name="externalBillingMeterRef" type="string">
  The external billing meter reference in the payment provider.
</ResponseField>

<ResponseField name="name" type="string">
  The fee name.
</ResponseField>

<ResponseField name="description" type="string">
  The fee description.
</ResponseField>

<ResponseField name="eventName" type="string">
  The event name that triggers this fee.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code (e.g. `BRL`, `USD`).
</ResponseField>

<ResponseField name="status" type="string">
  The fee status. One of: `ACTIVE`, `INACTIVE`.
</ResponseField>

<ResponseField name="createdBy" type="string">
  The user ID who created the fee.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedBy" type="string">
  The user ID who last updated the fee.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
