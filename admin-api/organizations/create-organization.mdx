---
title: "Create Organization"
api: "POST /organizations"
description: "Create an organization."
---

<Info>Requires `organizations:write` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Request Body

<ParamField body="name" type="string" required>Organization name. Max 255 chars.</ParamField>
<ParamField body="email" type="string" required>Billing email. Must be unique.</ParamField>
<ParamField body="currency" type="string" required>One of: `USD`, `BRL`, `EUR`.</ParamField>
<ParamField body="billingThresholdId" type="string" required>Must match org currency.</ParamField>
<ParamField body="phone" type="string">Phone number.</ParamField>
<ParamField body="address" type="object">
  Billing address.
  <Expandable title="address properties">
    <ParamField body="line1" type="string">Street address.</ParamField>
    <ParamField body="line2" type="string">Additional info.</ParamField>
    <ParamField body="city" type="string">City.</ParamField>
    <ParamField body="state" type="string">State/province.</ParamField>
    <ParamField body="postalCode" type="string">Postal code.</ParamField>
    <ParamField body="country" type="string">Country code (ISO 3166-1 alpha-2).</ParamField>
  </Expandable>
</ParamField>
<ParamField body="tax" type="object">
  Tax information.
  <Expandable title="tax properties">
    <ParamField body="taxId" type="string">Tax ID.</ParamField>
    <ParamField body="taxType" type="string">One of: `CPF`, `CNPJ`.</ParamField>
  </Expandable>
</ParamField>

<ResponseField name="organizationId" type="string">Organization UUID.</ResponseField>
<ResponseField name="billingThresholdId" type="string">Billing threshold ID.</ResponseField>
<ResponseField name="subscriptionId" type="string">The active subscription ID.</ResponseField>
<ResponseField name="iamExternalRef" type="string">Identity provider external reference.</ResponseField>
<ResponseField name="billingExternalRef" type="string">Payment provider external reference.</ResponseField>
<ResponseField name="name" type="string">Organization name.</ResponseField>
<ResponseField name="email" type="string">Billing email.</ResponseField>
<ResponseField name="currency" type="string">Currency code.</ResponseField>
<ResponseField name="phone" type="string">Phone number.</ResponseField>
<ResponseField name="address" type="object">Billing address.</ResponseField>
<ResponseField name="taxId" type="string">Tax ID.</ResponseField>
<ResponseField name="taxType" type="string">Tax type.</ResponseField>
<ResponseField name="offboardingStatus" type="string">The organization offboarding status.</ResponseField>
<ResponseField name="createdAt" type="string">Creation timestamp.</ResponseField>
<ResponseField name="updatedAt" type="string">Last update timestamp.</ResponseField>
