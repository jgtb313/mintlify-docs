---
title: "List Invoices"
api: "GET /invoices"
description: "List invoices."
---
<Info>Requires `invoice:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Query Parameters

<ParamField query="page" type="integer">Page number.</ParamField>
<ParamField query="limit" type="integer">Results per page.</ParamField>
<ParamField query="organizationId" type="string">Filter by organization ID.</ParamField>
<ParamField query="paymentMethod" type="string">Filter by payment method.</ParamField>
<ParamField query="status" type="string">Filter by status.</ParamField>
<ParamField query="type" type="string">Filter by type.</ParamField>
<ParamField query="sortBy" type="string">Sort field.</ParamField>
<ParamField query="sortOrder" type="string">Sort direction.</ParamField>

## Response

<ResponseField name="data" type="array">
  List of invoices.
  <Expandable title="invoice properties">
    <ResponseField name="invoiceId" type="string">Invoice UUID.</ResponseField>
    <ResponseField name="organizationId" type="string">Organization UUID.</ResponseField>
    <ResponseField name="subscriptionId" type="string">Subscription ID.</ResponseField>
    <ResponseField name="externalRef" type="string">External reference.</ResponseField>
    <ResponseField name="type" type="string">Invoice type.</ResponseField>
    <ResponseField name="paymentMethod" type="string">Payment method.</ResponseField>
    <ResponseField name="card" type="object">Card details (when payment method is CARD).</ResponseField>
    <ResponseField name="boleto" type="object">Boleto details (when payment method is BOLETO).</ResponseField>
    <ResponseField name="currency" type="string">Currency code.</ResponseField>
    <ResponseField name="totalAmount" type="integer">Total amount in cents.</ResponseField>
    <ResponseField name="discountAmount" type="integer">Discount amount in cents.</ResponseField>
    <ResponseField name="payableAmount" type="integer">Payable amount in cents.</ResponseField>
    <ResponseField name="discounts" type="array">Applied discounts.</ResponseField>
    <ResponseField name="lines" type="array">Invoice line items.</ResponseField>
    <ResponseField name="billingThreshold" type="object">Billing threshold (nullable).</ResponseField>
    <ResponseField name="status" type="string">Invoice status.</ResponseField>
    <ResponseField name="issuedAt" type="string">Issue timestamp.</ResponseField>
    <ResponseField name="dueDate" type="string">Due date.</ResponseField>
    <ResponseField name="paidAt" type="string">Payment timestamp.</ResponseField>
    <ResponseField name="createdAt" type="string">Creation timestamp.</ResponseField>
    <ResponseField name="updatedAt" type="string">Last update timestamp.</ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="meta" type="object">
  Pagination metadata.
  <Expandable title="meta properties">
    <ResponseField name="page" type="integer">Current page number.</ResponseField>
    <ResponseField name="limit" type="integer">Results per page.</ResponseField>
    <ResponseField name="totalItems" type="integer">Total number of items.</ResponseField>
    <ResponseField name="totalPages" type="integer">Total number of pages.</ResponseField>
  </Expandable>
</ResponseField>
