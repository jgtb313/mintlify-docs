---
title: "The Voucher Object"
---

<ResponseField name="voucherId" type="string">
  The voucher unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID.
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for the voucher.
</ResponseField>

<ResponseField name="name" type="string">
  The voucher display name.
</ResponseField>

<ResponseField name="amount" type="integer">
  The voucher total amount in cents.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code (e.g. `BRL`, `USD`).
</ResponseField>

<ResponseField name="effectiveAt" type="string|null">
  Timestamp when the voucher becomes effective (ISO 8601).
</ResponseField>

<ResponseField name="expiresAt" type="string|null">
  Timestamp when the voucher expires (ISO 8601).
</ResponseField>

<ResponseField name="amountRedeemed" type="integer">
  The amount already redeemed in cents.
</ResponseField>

<ResponseField name="status" type="string">
  The voucher status. One of: `PENDING`, `ACTIVE`, `DEPLETED`, `EXPIRED`.
</ResponseField>

<ResponseField name="createdBy" type="string">
  The actor who created the voucher.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedBy" type="string">
  The actor who last updated the voucher.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>

<ResponseField name="deletedBy" type="string">
  The actor who deleted the voucher.
</ResponseField>

<ResponseField name="deletedAt" type="string">
  Deletion timestamp (ISO 8601).
</ResponseField>
