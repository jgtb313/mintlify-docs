---
title: "The Coupon Usage Object"
---

<ResponseField name="couponUsageId" type="string">
  The coupon usage unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="couponId" type="string">
  The coupon ID associated with this usage.
</ResponseField>

<ResponseField name="invoiceId" type="string">
  The invoice ID associated with this usage.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID that redeemed the coupon.
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for this usage.
</ResponseField>

<ResponseField name="amount" type="integer">
  The discount amount applied in cents.
</ResponseField>

<ResponseField name="status" type="string">
  The coupon usage status.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
