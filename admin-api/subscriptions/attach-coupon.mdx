---
title: "Attach Coupon"
api: "POST /organizations/{organizationId}/subscriptions/attach-coupon"
description: "Attach a coupon to a subscription."
---

Attaches a coupon to the organization's active subscription.

<Info>Requires `coupon:attach` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

## Request Body

<ParamField body="couponCode" type="string" required>
  The coupon code to attach.
</ParamField>

## Response

<ResponseField name="subscriptionId" type="string">
  The subscription ID.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="planId" type="string">
  The plan ID.
</ResponseField>

<ResponseField name="planIntervalId" type="string">
  The plan interval ID.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code.
</ResponseField>

<ResponseField name="status" type="string">
  The subscription status.
</ResponseField>

<ResponseField name="coupons" type="array">
  Attached subscription coupons.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
