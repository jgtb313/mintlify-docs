---
title: "Get Offboarding Request"
api: "GET /organizations/{organizationId}/offboarding-requests/{offboardingRequestId}"
description: "Retrieve an offboarding request with its events."
---

<Info>Requires `offboarding:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>Organization UUID.</ParamField>
<ParamField path="offboardingRequestId" type="string" required>Offboarding request ID (UUIDv7).</ParamField>

## Response

<ResponseField name="offboardingRequestId" type="string">Offboarding request UUID.</ResponseField>
<ResponseField name="organizationId" type="string">Organization UUID.</ResponseField>
<ResponseField name="action" type="string">The offboarding action.</ResponseField>
<ResponseField name="reason" type="string">The primary reason.</ResponseField>
<ResponseField name="reasonDetail" type="string|null">Additional detail.</ResponseField>
<ResponseField name="notes" type="string|null">Free-text notes.</ResponseField>
<ResponseField name="campaignMode" type="string|null">The campaign mode.</ResponseField>
<ResponseField name="billingMode" type="string|null">The billing mode.</ResponseField>
<ResponseField name="billingBehavior" type="string|null">The billing behavior.</ResponseField>
<ResponseField name="retentionDeadline" type="string|null">The retention deadline (ISO 8601).</ResponseField>
<ResponseField name="retentionResolution" type="string|null">The retention resolution.</ResponseField>
<ResponseField name="status" type="string">The offboarding request status.</ResponseField>
<ResponseField name="requestedBy" type="string">User UUID who requested.</ResponseField>
<ResponseField name="requestedAt" type="string">Request timestamp (ISO 8601).</ResponseField>
<ResponseField name="resolvedBy" type="string|null">User UUID who resolved.</ResponseField>
<ResponseField name="resolvedAt" type="string|null">Resolution timestamp (ISO 8601).</ResponseField>
<ResponseField name="cancelledBy" type="string|null">User UUID who cancelled.</ResponseField>
<ResponseField name="cancelledBySystem" type="boolean|null">Whether cancelled by system.</ResponseField>
<ResponseField name="cancelledReason" type="string|null">Cancellation reason.</ResponseField>
<ResponseField name="cancelledAt" type="string|null">Cancellation timestamp (ISO 8601).</ResponseField>
<ResponseField name="cancellationDetails" type="object|null">Additional cancellation details.</ResponseField>
<ResponseField name="createdAt" type="string">Creation timestamp (ISO 8601).</ResponseField>
<ResponseField name="updatedAt" type="string">Last update timestamp (ISO 8601).</ResponseField>
<ResponseField name="events" type="array">
  The offboarding events for this request.
  <Expandable title="event properties">
    <ResponseField name="offboardingEventId" type="string">Event UUID.</ResponseField>
    <ResponseField name="eventType" type="string">The type of event.</ResponseField>
    <ResponseField name="createdBy" type="string">User UUID who triggered the event.</ResponseField>
    <ResponseField name="createdAt" type="string">Event timestamp (ISO 8601).</ResponseField>
  </Expandable>
</ResponseField>
