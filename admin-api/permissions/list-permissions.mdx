---
title: "List Permissions"
api: "GET /permissions"
description: "List permissions."
---

<Info>Requires `permissions:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Query Parameters

<ParamField query="limit" type="integer">Results per page.</ParamField>
<ParamField query="previousCursor" type="string">Cursor for previous page.</ParamField>
<ParamField query="nextCursor" type="string">Cursor for next page.</ParamField>

## Response

<ResponseField name="data" type="array">
  List of permissions.
  <Expandable title="permission properties">
    <ResponseField name="permissionId" type="integer">Permission ID.</ResponseField>
    <ResponseField name="slug" type="string">Permission slug.</ResponseField>
    <ResponseField name="resource" type="string">Resource name.</ResponseField>
    <ResponseField name="action" type="string">Action name.</ResponseField>
    <ResponseField name="name" type="string">Permission name.</ResponseField>
    <ResponseField name="description" type="string">Permission description.</ResponseField>
    <ResponseField name="system" type="boolean">Whether it is a system permission.</ResponseField>
    <ResponseField name="createdAt" type="string">Creation timestamp.</ResponseField>
    <ResponseField name="updatedAt" type="string">Last update timestamp.</ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="meta" type="object">
  Pagination metadata.
  <Expandable title="meta properties">
    <ResponseField name="hasMore" type="boolean">Whether more results are available.</ResponseField>
    <ResponseField name="limit" type="integer">Results per page.</ResponseField>
  </Expandable>
</ResponseField>
