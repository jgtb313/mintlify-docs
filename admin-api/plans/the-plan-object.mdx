---
title: "The Plan Object"
---

<ResponseField name="planId" type="string">
  The plan unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for the plan in the payment provider.
</ResponseField>

<ResponseField name="name" type="string">
  The plan name.
</ResponseField>

<ResponseField name="description" type="string">
  The plan description.
</ResponseField>

<ResponseField name="features" type="object[]">
  The list of features included in the plan.
  <Expandable title="feature properties">
    <ResponseField name="description" type="string">
      The feature description.
    </ResponseField>
    <ResponseField name="type" type="string">
      The feature inclusion type. One of: `INCLUDE`, `NOT_INCLUDE`.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="intervals" type="object[]">
  The billing intervals available for the plan.
  <Expandable title="interval properties">
    <ResponseField name="planIntervalId" type="string">
      The interval unique identifier (UUIDv7).
    </ResponseField>
    <ResponseField name="planId" type="string">
      The parent plan ID (UUIDv7).
    </ResponseField>
    <ResponseField name="externalRef" type="string">
      The external reference for this interval in the payment provider.
    </ResponseField>
    <ResponseField name="interval" type="string">
      The billing interval. One of: `MONTHLY`, `YEARLY`.
    </ResponseField>
    <ResponseField name="amount" type="number">
      The price amount for this interval.
    </ResponseField>
    <ResponseField name="currency" type="string">
      The currency code (e.g., `USD`).
    </ResponseField>
    <ResponseField name="status" type="string">
      The interval status. One of: `ACTIVE`, `INACTIVE`.
    </ResponseField>
    <ResponseField name="createdBy" type="string">
      The user ID who created this interval.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Creation timestamp (ISO 8601).
    </ResponseField>
    <ResponseField name="updatedBy" type="string">
      The user ID who last updated this interval.
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Last update timestamp (ISO 8601).
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="highlight" type="boolean">
  Whether the plan is highlighted in the pricing page.
</ResponseField>

<ResponseField name="status" type="string">
  The plan status. One of: `ACTIVE`, `INACTIVE`.
</ResponseField>

<ResponseField name="createdBy" type="string">
  The user ID who created the plan.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedBy" type="string">
  The user ID who last updated the plan.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
