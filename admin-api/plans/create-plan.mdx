---
title: "Create Plan"
api: "POST /plans"
description: "Create a plan."
---

<Info>Requires `plan:write` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Request Body

<ParamField body="name" type="string" required>Plan name. Max 255 chars.</ParamField>
<ParamField body="description" type="string" required>Plan description.</ParamField>
<ParamField body="highlight" type="boolean">Whether to highlight this plan. Default: false.</ParamField>
<ParamField body="features" type="array" required>
  Plan features list.
  <Expandable title="feature properties">
    <ParamField body="description" type="string" required>Feature description.</ParamField>
    <ParamField body="type" type="string" required>One of: `INCLUDE`, `NOT_INCLUDE`, `DEFAULT`.</ParamField>
  </Expandable>
</ParamField>
<ParamField body="intervals" type="array" required>
  Billing intervals.
  <Expandable title="interval properties">
    <ParamField body="interval" type="string" required>One of: `MONTHLY`, `YEARLY`.</ParamField>
    <ParamField body="amount" type="integer" required>Price in cents.</ParamField>
    <ParamField body="currency" type="string" required>One of: `USD`, `BRL`, `EUR`.</ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="planId" type="string">Plan ID.</ResponseField>
<ResponseField name="externalRef" type="string">External reference.</ResponseField>
<ResponseField name="name" type="string">Plan name.</ResponseField>
<ResponseField name="description" type="string">Plan description.</ResponseField>
<ResponseField name="features" type="array">Plan features.</ResponseField>
<ResponseField name="intervals" type="array">Billing intervals.</ResponseField>
<ResponseField name="highlight" type="boolean">Whether the plan is highlighted.</ResponseField>
<ResponseField name="status" type="string">Plan status.</ResponseField>
<ResponseField name="createdBy" type="string">ID of the user who created the plan.</ResponseField>
<ResponseField name="createdAt" type="string">Creation timestamp.</ResponseField>
<ResponseField name="updatedBy" type="string">ID of the user who last updated the plan.</ResponseField>
<ResponseField name="updatedAt" type="string">Last update timestamp.</ResponseField>
