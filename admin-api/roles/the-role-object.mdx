---
title: "The Role Object"
---

<ResponseField name="roleId" type="string">
  The role unique identifier.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID.
</ResponseField>

<ResponseField name="slug" type="string">
  The role slug.
</ResponseField>

<ResponseField name="name" type="string">
  The role display name.
</ResponseField>

<ResponseField name="description" type="string|null">
  The role description. `null` if not provided.
</ResponseField>

<ResponseField name="type" type="string">
  The role type. One of: `ENVIRONMENT_ROLE`, `ORGANIZATION_ROLE`.
</ResponseField>

<ResponseField name="permissions" type="string[]">
  Array of permission slugs assigned to the role.
</ResponseField>

<ResponseField name="inherited" type="boolean">
  Whether the role is inherited from a parent scope.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
