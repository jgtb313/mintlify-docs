---
title: "The User Object"
---

<ResponseField name="userId" type="string">
  The user unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for the user in the identity provider.
</ResponseField>

<ResponseField name="email" type="string">
  The user email address.
</ResponseField>

<ResponseField name="firstName" type="string">
  The user first name.
</ResponseField>

<ResponseField name="lastName" type="string">
  The user last name.
</ResponseField>

<ResponseField name="locale" type="string|null">
  The user preferred locale (e.g. `en`, `pt-BR`). Null if not set.
</ResponseField>

<ResponseField name="avatar" type="string|null">
  The URL of the user profile picture. Null if not set.
</ResponseField>

<ResponseField name="lastSignInAt" type="string|null">
  Timestamp of the user last sign-in (ISO 8601). Null if the user has never signed in.
</ResponseField>

<ResponseField name="emailVerified" type="boolean">
  Whether the user email has been verified.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
