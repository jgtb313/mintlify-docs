---
title: "The Invoice Object"
---

<ResponseField name="invoiceId" type="string">
  The invoice unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID.
</ResponseField>

<ResponseField name="subscriptionId" type="string|null">
  The associated subscription ID (UUIDv7). Null for standalone invoices.
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for the invoice.
</ResponseField>

<ResponseField name="type" type="string">
  The invoice type. One of: `PLAN`, `FEE`.
</ResponseField>

<ResponseField name="paymentMethod" type="string|null">
  The payment method used.
</ResponseField>

<ResponseField name="card" type="object|null">
  Card details when the payment method is a card.
  <Expandable title="card properties">
    <ResponseField name="brand" type="string">
      The card brand (e.g. `visa`, `mastercard`).
    </ResponseField>
    <ResponseField name="last4" type="string">
      The last four digits of the card number.
    </ResponseField>
    <ResponseField name="expMonth" type="integer">
      The card expiration month.
    </ResponseField>
    <ResponseField name="expYear" type="integer">
      The card expiration year.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="boleto" type="object|null">
  Boleto details when the payment method is a boleto.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code (e.g. `BRL`, `USD`).
</ResponseField>

<ResponseField name="totalAmount" type="integer">
  The total amount in cents.
</ResponseField>

<ResponseField name="discountAmount" type="integer">
  The discount amount in cents.
</ResponseField>

<ResponseField name="payableAmount" type="integer">
  The payable amount in cents (total minus discounts).
</ResponseField>

<ResponseField name="discounts" type="array">
  The discounts applied to this invoice.
</ResponseField>

<ResponseField name="lines" type="array">
  The invoice line items.
  <Expandable title="line item properties">
    <ResponseField name="id" type="string">
      The line item unique identifier.
    </ResponseField>
    <ResponseField name="name" type="string">
      The line item description.
    </ResponseField>
    <ResponseField name="unitPrice" type="integer">
      The unit price in cents.
    </ResponseField>
    <ResponseField name="quantity" type="integer">
      The quantity.
    </ResponseField>
    <ResponseField name="amount" type="integer">
      The total amount for this line item in cents.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="status" type="string">
  The invoice status. One of: `DRAFT`, `OPEN`, `PAID`, `PAYMENT_FAILED`, `REFUNDED`, `PARTIALLY_REFUNDED`, `DISPUTED`, `CHARGEBACK`, `CHARGEBACK_RESOLVED`, `UNCOLLECTIBLE`, `VOID`, `EXPIRED`.
</ResponseField>

<ResponseField name="issuedAt" type="string">
  Timestamp when the invoice was issued (ISO 8601).
</ResponseField>

<ResponseField name="dueDate" type="string">
  The invoice due date (ISO 8601).
</ResponseField>

<ResponseField name="paidAt" type="string|null">
  Timestamp when the invoice was paid (ISO 8601).
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
