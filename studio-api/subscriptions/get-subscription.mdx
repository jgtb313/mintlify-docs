---
title: "Get Subscription"
api: "GET /organizations/{organizationId}/subscriptions"
description: "Retrieve the organization's subscription."
---

Returns the organization's active subscription.

<Info>Requires `subscription:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

## Response

<ResponseField name="subscriptionId" type="string">
  The subscription unique identifier.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="planId" type="string">
  The plan ID.
</ResponseField>

<ResponseField name="planIntervalId" type="string">
  The plan interval ID.
</ResponseField>

<ResponseField name="externalPlanRef" type="string | null">
  External reference for the plan subscription.
</ResponseField>

<ResponseField name="externalFeeRef" type="string | null">
  External reference for fees.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code (e.g. `BRL`, `USD`).
</ResponseField>

<ResponseField name="status" type="string">
  The subscription status. One of: `ACTIVE`, `PAST_DUE`, `PAUSED`, `CANCELLATION_PENDING`, `CANCELLED`.
</ResponseField>

<ResponseField name="pastDueReason" type="string | null">
  Reason the subscription is past due.
</ResponseField>

<ResponseField name="pastDueAt" type="string | null">
  ISO 8601 timestamp when the subscription became past due.
</ResponseField>

<ResponseField name="pausedBy" type="string | null">
  User who paused the subscription.
</ResponseField>

<ResponseField name="pausedAt" type="string | null">
  ISO 8601 timestamp when the subscription was paused.
</ResponseField>

<ResponseField name="cancelledBy" type="string | null">
  User who cancelled the subscription.
</ResponseField>

<ResponseField name="cancelledAt" type="string | null">
  ISO 8601 timestamp when the subscription was cancelled.
</ResponseField>

<ResponseField name="coupons" type="array">
  Applied coupons.
</ResponseField>

<ResponseField name="createdBy" type="string">
  User who created the subscription.
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 creation timestamp.
</ResponseField>

<ResponseField name="updatedBy" type="string">
  User who last updated the subscription.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 last-update timestamp.
</ResponseField>
