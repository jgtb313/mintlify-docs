---
title: "Get Upcoming Invoice"
api: "GET /organizations/{organizationId}/subscriptions/upcoming-invoice"
description: "Preview the upcoming invoice."
---

Returns a preview of the upcoming invoice.

<Info>Requires `subscription:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

## Response

<ResponseField name="invoiceId" type="string">
  The invoice unique identifier.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="subscriptionId" type="string">
  The subscription ID.
</ResponseField>

<ResponseField name="paymentMethod" type="string|null">
  The payment method type.
</ResponseField>

<ResponseField name="card" type="object|null">
  Card details, if applicable.
</ResponseField>

<ResponseField name="boleto" type="object|null">
  Boleto details, if applicable.
</ResponseField>

<ResponseField name="currency" type="string">
  The currency code.
</ResponseField>

<ResponseField name="totalAmount" type="integer">
  The total amount in cents.
</ResponseField>

<ResponseField name="discountAmount" type="integer">
  The discount amount in cents.
</ResponseField>

<ResponseField name="payableAmount" type="integer">
  The payable amount in cents.
</ResponseField>

<ResponseField name="lines" type="array">
  The invoice line items.
</ResponseField>

<ResponseField name="discounts" type="array">
  The applied discounts.
</ResponseField>

<ResponseField name="billingThreshold" type="object|null">
  The billing threshold associated with the invoice.
</ResponseField>

<ResponseField name="status" type="string">
  The invoice status.
</ResponseField>

<ResponseField name="dueDate" type="string">
  The due date (ISO 8601).
</ResponseField>
