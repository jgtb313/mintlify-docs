---
title: "Create Membership"
api: "POST /organizations/{organizationId}/memberships"
description: "Create a membership."
---

Creates a new membership.

<Info>Requires `memberships:write` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

## Request Body

<ParamField body="userId" type="string" required>
  The user ID to add as a member.
</ParamField>

<ParamField body="roleSlug" type="string">
  The role slug to assign.
</ParamField>

## Response

<ResponseField name="membershipId" type="string">
  The membership unique identifier.
</ResponseField>

<ResponseField name="userId" type="string">
  The user ID.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="role" type="object">
  The assigned role.
</ResponseField>

<ResponseField name="status" type="string">
  The membership status.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
