---
title: "The Membership Object"
---

<ResponseField name="membershipId" type="string">
  The membership unique identifier.
</ResponseField>

<ResponseField name="userId" type="string">
  The user ID (UUIDv7).
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID (UUIDv7).
</ResponseField>

<ResponseField name="role" type="object">
  The role assigned to the member.
  <Expandable title="role properties">
    <ResponseField name="slug" type="string">
      The role slug.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="status" type="string">
  The membership status. One of: `ACTIVE`, `INACTIVE`, `PENDING`.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
