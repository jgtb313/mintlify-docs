---
title: "Get Membership"
api: "GET /organizations/{organizationId}/memberships/{membershipId}"
description: "Retrieve a membership by its ID."
---

Returns a membership by its ID.

<Info>Requires `memberships:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

<ParamField path="membershipId" type="string" required>
  The membership ID.
</ParamField>

## Response

<ResponseField name="membershipId" type="string">
  The membership ID.
</ResponseField>

<ResponseField name="userId" type="string">
  The user ID.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="role" type="object">
  The assigned role.
</ResponseField>

<ResponseField name="status" type="string">
  The membership status.
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 creation timestamp.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 last-update timestamp.
</ResponseField>
