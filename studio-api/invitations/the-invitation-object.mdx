---
title: "The Invitation Object"
---

<ResponseField name="invitationId" type="string">
  The invitation unique identifier.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID (UUIDv7) the invitation belongs to.
</ResponseField>

<ResponseField name="inviterUserId" type="string">
  The user ID (UUIDv7) of the member who sent the invitation.
</ResponseField>

<ResponseField name="acceptedUserId" type="string|null">
  The user ID (UUIDv7) of the member who accepted the invitation. `null` if not yet accepted.
</ResponseField>

<ResponseField name="email" type="string">
  The email address of the invited user.
</ResponseField>

<ResponseField name="token" type="string">
  The unique token associated with the invitation.
</ResponseField>

<ResponseField name="acceptInvitationUrl" type="string">
  The URL the invited user can use to accept the invitation.
</ResponseField>

<ResponseField name="roleSlug" type="string|null">
  The role slug assigned upon acceptance. `null` if not specified.
</ResponseField>

<ResponseField name="state" type="string">
  The invitation state. One of: `PENDING`, `ACCEPTED`, `EXPIRED`, `REVOKED`.
</ResponseField>

<ResponseField name="expiresAt" type="string">
  Expiration timestamp (ISO 8601).
</ResponseField>

<ResponseField name="acceptedAt" type="string|null">
  Timestamp when the invitation was accepted (ISO 8601). `null` if not yet accepted.
</ResponseField>

<ResponseField name="revokedAt" type="string|null">
  Timestamp when the invitation was revoked (ISO 8601). `null` if not revoked.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
