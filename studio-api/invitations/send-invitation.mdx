---
title: "Send Invitation"
api: "POST /organizations/{organizationId}/invitations"
description: "Send an invitation."
---

Sends an invitation.

<Info>Requires `invitations:write` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

## Request Body

<ParamField body="email" type="string" required>
  Email address to invite.
</ParamField>

<ParamField body="roleSlug" type="string">
  Optional role slug to assign upon acceptance.
</ParamField>

<ParamField body="expiresInDays" type="integer">
  Number of days until the invitation expires.
</ParamField>

<ParamField body="inviterUserId" type="string">
  The user ID of the inviter.
</ParamField>

## Response

<ResponseField name="invitationId" type="string">
  The invitation ID.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID.
</ResponseField>

<ResponseField name="inviterUserId" type="string">
  The user ID of the inviter.
</ResponseField>

<ResponseField name="acceptedUserId" type="string | null">
  The user ID of the member who accepted the invitation.
</ResponseField>

<ResponseField name="email" type="string">
  The invited email address.
</ResponseField>

<ResponseField name="token" type="string">
  The invitation token.
</ResponseField>

<ResponseField name="acceptInvitationUrl" type="string">
  URL for accepting the invitation.
</ResponseField>

<ResponseField name="roleSlug" type="string | null">
  The role slug assigned upon acceptance.
</ResponseField>

<ResponseField name="state" type="string">
  The invitation state.
</ResponseField>

<ResponseField name="expiresAt" type="string">
  ISO 8601 expiration timestamp.
</ResponseField>

<ResponseField name="acceptedAt" type="string | null">
  ISO 8601 timestamp when the invitation was accepted.
</ResponseField>

<ResponseField name="revokedAt" type="string | null">
  ISO 8601 timestamp when the invitation was revoked.
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 creation timestamp.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 last-update timestamp.
</ResponseField>
