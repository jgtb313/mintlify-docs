---
title: "The Voucher Usage Object"
---

<ResponseField name="voucherUsageId" type="string">
  The voucher usage unique identifier (UUIDv7).
</ResponseField>

<ResponseField name="voucherId" type="string">
  The voucher ID associated with this usage.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization ID that used the voucher.
</ResponseField>

<ResponseField name="discountAmount" type="integer">
  The discount amount applied in cents.
</ResponseField>

<ResponseField name="externalRef" type="string">
  The external reference for this usage.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Creation timestamp (ISO 8601).
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Last update timestamp (ISO 8601).
</ResponseField>
