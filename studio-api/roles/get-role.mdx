---
title: "Get Role"
api: "GET /organizations/{organizationId}/roles/{slug}"
description: "Retrieve a role by its slug."
---

Returns a role by its slug.

<Info>Requires `roles:read` permission.</Info>

## Headers

<ParamField header="Authorization" type="string" required>
  Bearer token.
</ParamField>

## Path Parameters

<ParamField path="organizationId" type="string" required>
  The organization UUID.
</ParamField>

<ParamField path="slug" type="string" required>
  The role slug.
</ParamField>

## Response

<ResponseField name="roleId" type="string">
  The role ID.
</ResponseField>

<ResponseField name="organizationId" type="string">
  The organization UUID. Present only for organization roles.
</ResponseField>

<ResponseField name="slug" type="string">
  Unique role identifier.
</ResponseField>

<ResponseField name="name" type="string">
  Display name.
</ResponseField>

<ResponseField name="description" type="string">
  Role description.
</ResponseField>

<ResponseField name="type" type="string">
  Role type (`ENVIRONMENT_ROLE` or `ORGANIZATION_ROLE`).
</ResponseField>

<ResponseField name="permissions" type="string[]">
  Array of permission slugs.
</ResponseField>

<ResponseField name="inherited" type="boolean">
  Whether the role is inherited from the environment.
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 creation timestamp.
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO 8601 last-update timestamp.
</ResponseField>
